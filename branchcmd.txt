create branch
git checkout -b devlop

change branch
git checkout master

view branch list
git branch

merge branch with master
git merge devlop

view branch's last commit and massage
git branch -v

git branch --merged (view merged branch)
git branch --no-merged

delete branch 
git branch -d devlop
git branch -D devlop (delete unmerged branch)

delete pushed branch 
git push -d origin devlop
